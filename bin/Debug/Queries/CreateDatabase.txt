CREATE DATABASE TestDB;

CREATE TABLE Customer(
Customer_ID INT PRIMARY KEY IDENTITY(1,1),
FirstName varchar(50),
LastName varchar(50),
EmailAddres varchar(50),
PhoneNumber nvarchar(50),
ContactPerson varchar(50));
CREATE TABLE Payments(
Payments_ID INT PRIMARY KEY IDENTITY(1,1),
Customer_ID int FOREIGN KEY REFERENCES Customer(Customer_ID),
FirstName varchar(50),
LastName varchar(50),
CreditCard varchar(50),
Date Date,
CardNumber varchar(16),
CVV varchar(3),
ExpirationDate Date
);
CREATE TABLE Room(
Room_ID INT PRIMARY KEY IDENTITY(1,1),
RoomNumber int,
CustomerMax int,
Available bit, 
Cost money
);
CREATE TABLE Booking(
Booking_ID INT PRIMARY KEY IDENTITY(1,1),
Customer_ID int FOREIGN KEY REFERENCES Customer(Customer_ID),
Payments_ID int FOREIGN KEY REFERENCES Payments(Payments_ID), 
Room_ID int FOREIGN KEY REFERENCES Room(Room_ID),
CheckInDate Date,
CheckOutDate Date,
NumberOfGuests INT,
BookingDate Date,
DiscountCode varchar(50),
TotalCost varchar(50) 
);

CREATE TABLE UserLogins(
Id, PRIMARY KEY IDENTITY(1,),
UserName varchar(50),
Password varchar(50),
Roles varchar(50));
